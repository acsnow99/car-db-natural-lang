Which city has the most rides?
SELECT City.cityCode, COUNT(*) AS num_rides
FROM RideRecord
JOIN Car ON RideRecord.VIN = Car.VIN
JOIN City ON Car.cityCode = City.cityCode
GROUP BY City.cityCode
ORDER BY num_rides DESC
LIMIT 1;
